\begin{column}{0.3\hsize}
\vspace{-10pt}
\begin{align*}
h_0(x)
&=
(2x+1)(x-1)^2
\\
h_0(0)&=1,\quad h_0(1)=0
%\\
%h_0'(0)&=h_0'(1)=0
\\[5pt]
h_1(x)
&=
x^2(2x-3)
\\
h_1(0)&=0,\quad h_1(1)=1
%\\
%h_1'(0)&=h_1'(1)=0
\\[5pt]
h_i'(0)&=h_i'(1)=0
\\[15pt]
h_0^1(x)
&=
x(x-1)^2
\\
h_0^{1\prime}(0)&=0,\quad h_0^{1\prime}(1)=1
\\[5pt]
h_1^1(x)
&=
x^2(x-1)
\\
h_1^{1\prime}(0)&=1,\quad h_1^{1\prime}(1)=0
\\[5pt]
h_i^1(0)&=h_i^1(1)=0
\end{align*}
\end{column}
